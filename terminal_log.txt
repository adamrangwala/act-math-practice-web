rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // Users can only read or update their own document.
    // Any authenticated user can create their own user document.
    match /users/{userId} {
      allow read, update, delete: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null;
    }

    // A user can only read or write their own stats document.
    match /userStats/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // A user can only read or write their own progress documents.
    match /userSubcategoryProgress/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Questions are read-only for any authenticated user.
    // No one can write to the questions collection from the client-side.
    // This protects your question data from being modified or deleted.
    match /questions/{questionId} {
      allow read: if request.auth != null;
      allow write: if false;
    }

    // --- NEW RULE FOR FEEDBACK ---
    // Any authenticated user can create (submit) feedback.
    // No one can read, update, or delete feedback from the client-side.
    match /feedback/{feedbackId} {
      allow create: if request.auth != null;
      allow read, update, delete: if false;
    }
  }
}
